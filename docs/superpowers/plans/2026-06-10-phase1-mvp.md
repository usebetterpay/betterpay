**Architecture:** Monorepo with plugin-first architecture. Core package provides better-call router, plugin system, provider interface with priority-based selection and circuit breaker. Provider adapters extracted from wabase production code. Webhook pipeline with idempotency and signature verification.

**Tech Stack:** TypeScript, pnpm workspace, Turborepo, Biome, Vitest, better-call, drizzle-orm, pg

```typescript
export interface PayEndpoint {

  // Will be defined later
  (ctx: PayContext): Promise<void>;
}

export interface HookContext {

  method: string;
  [key: string]: unknown;
}

export interface RawError {
  code: string;
  message: string;
}

export interface BetterPayPlugin {
  id: string;
  version?: string;

  // Lifecycle
  init?: (ctx: PayContext) => Promise<void>;

  // HTTP
  endpoints?: Record<string, PayEndpoint>;
  middlewares?: Array<{ path: string; middleware: PayMiddleware }>;
  onRequest?: (req: Request, ctx: PayContext) => Promise<{ response: Response } | void>;
  onResponse?: (res: Response, ctx: PayContext) => Promise<void>;

  // Hooks
  hooks?: {
    before?: Array<{ matcher: (ctx: HookContext) => boolean; handler: PayMiddleware }>;
    after?: Array<{ matcher: (ctx: HookContext) => boolean; handler: PayMiddleware }>;
  };

  // Database
  schema?: Record<string, unknown>;
  migrations?: Record<string, unknown>;

  // Provider
  providers?: unknown[];
  defaultProvider?: string;

  // Notifications
  notificationChannels?: unknown[];

  // Rate Limiting
  rateLimit?: Array<{
    window: number;
    max: number;
    pathMatcher: (path: string) => boolean;
  }>;

  // Type Safety
  $ERROR_CODES?: Record<string, RawError>;
  $Infer?: Record<string, unknown>;
}
```

```typescript
// packages/core/src/provider/registry.ts
import type { PaymentProvider, PaymentMethod } from './interface';

interface ProviderWithPriority extends PaymentProvider {
  priority?: number;
}

export class ProviderRegistry {
  private providers = new Map<string, ProviderWithPriority>();

  register(provider: ProviderWithPriority): void {
    this.providers.set(provider.id, provider);
  }

  get(id: string): PaymentProvider | undefined {
    return this.providers.get(id);
  }

  list(): PaymentProvider[] {
    return Array.from(this.providers.values());
  }

  getDefault(): PaymentProvider {
    const providers = this.list();
    if (providers.length === 0) {
      throw new Error('No providers registered');
    }
    
    // Sort by priority (lower number = higher priority)
    const sorted = [...providers].sort((a, b) => {
      const priorityA = (a as ProviderWithPriority).priority ?? 999;
      const priorityB = (b as ProviderWithPriority).priority ?? 999;
      return priorityA - priorityB;
    });
    
    return sorted[0]!;
  }

  findByMethod(method: PaymentMethod): PaymentProvider[] {
    return this.list().filter(provider => 
      provider.paymentMethods.includes(method)
    );
  }

  selectForSubscribe(input: {
    paymentMethod?: PaymentMethod;
    amount?: number;
  }): PaymentProvider {
    let candidates: PaymentProvider[];
    
    if (input.paymentMethod) {
      candidates = this.findByMethod(input.paymentMethod);
      if (candidates.length === 0) {
        throw new Error(`No provider supports payment method: ${input.paymentMethod}`);
      }
    } else {
      candidates = this.list();
    }
    
    // Sort by priority
    const sorted = [...candidates].sort((a, b) => {
      const priorityA = (a as ProviderWithPriority).priority ?? 999;
      const priorityB = (b as ProviderWithPriority).priority ?? 999;
      return priorityA - priorityB;
    });
    
    return sorted[0]!;
  }
}
```
